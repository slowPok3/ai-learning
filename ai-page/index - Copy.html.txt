    function extractCourseTitle(markdown, fallback = "Curriculum") {
        const lines = markdown.split("\n");
        for (const line of lines) {
            const trimmed = line.trim();
            if (trimmed.startsWith("# ") && !trimmed.startsWith("## ")) {
                return trimmed.slice(2).trim();
            }
        }
        return fallback;
    }

    function parseMarkdownToData(mdStr) {
        const lines = mdStr.split('\n');
        const data = [];
        let currentDomain = null;
        let currentTopic = null;
        let topicCounter = 0;

        lines.forEach(line => {
            line = line.trim();
            if (!line) return;

            if (line.startsWith('# ') && !line.startsWith('## ')) {
                return;
            }
            else if (line.startsWith('## ') && !line.startsWith('### ')) {
                currentDomain = line.substring(3).trim();
                currentTopic = null; 
            }
            else if (line.startsWith('### ') && !line.startsWith('#### ')) {
                currentTopic = line.substring(4).trim();
                topicCounter++;
            }
            else if (line.startsWith('- ')) {
                if (currentDomain && currentTopic) {
                    const parts = line.substring(2).split(':');
                    const name = parts[0].trim();
                    const desc = parts.slice(1).join(':').trim() || '';

                    let topicEntry = data.find(
                        d => d.domain === currentDomain && d.topic === currentTopic
                    );
                    if (!topicEntry) {
                        topicEntry = {
                            domain: currentDomain,
                            topic: currentTopic,
                            topicId: 'T' + topicCounter,
                            subjects: []
                        };
                        data.push(topicEntry);
                    }

                    const subjectCounter = topicEntry.subjects.length + 1;
                    const subId = topicEntry.topicId + '-S' + subjectCounter;

                    topicEntry.subjects.push({
                        name,
                        desc,
                        id: subId,
                        domain: currentDomain,
                        topic: currentTopic
                    });
                }
            }
        });

        return data;
    }

    function openDomainDetailModal(domainName) {

        const domainNames = [...new Set(parsedCurriculumData.map(d => d.domain))];

    function nextView() { switchView(views[(currentViewIndex + 1) % views.length]); }
    function prevView() { switchView(views[(currentViewIndex - 1 + views.length) % views.length]); }

    // ─────────────────────────────────────────────────────────────────
    // VIEW 1: VECTOR SPACE / CONSTELLATION
    // ─────────────────────────────────────────────────────────────────
    function initVectorSpace() {
        const container = document.getElementById('vector-container');
        const scene  = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(60, window.innerWidth / (window.innerHeight - 60), 0.1, 2000);
        
        const homePosition = { x: 0, y: 100, z: 320 };
        camera.position.set(homePosition.x, homePosition.y, homePosition.z);

        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(window.innerWidth, window.innerHeight - 60);
        renderer.setPixelRatio(window.devicePixelRatio);
        container.appendChild(renderer.domElement);

        const controls = new THREE.OrbitControls(camera, renderer.domElement);
        controls.enableDamping  = true;
        controls.dampingFactor  = 0.05;
        controls.enablePan      = true;
        controls.minDistance    = 40;   
        controls.maxDistance    = 800;  

        scene.add(new THREE.AmbientLight(0xffffff, 0.4));
        const pointLight = new THREE.PointLight(0xffffff, 1.2);
        pointLight.position.set(0, 200, 0);
        scene.add(pointLight);

        const starGeo  = new THREE.BufferGeometry();
        const starVerts = [];
        for (let i = 0; i < 1500; i++) {
            starVerts.push((Math.random() - 0.5) * 1600, (Math.random() - 0.5) * 1600, (Math.random() - 0.5) * 1600);
        }
        starGeo.setAttribute('position', new THREE.Float32BufferAttribute(starVerts, 3));
        const starField = new THREE.Points(starGeo, new THREE.PointsMaterial({ color: 0xffffff, size: 0.8, transparent: true, opacity: 0.5 }));
        scene.add(starField);

        function createNodeMesh(size, colorHex, emissiveHex, isMoon = false) {
            const geo  = new THREE.SphereGeometry(size, 32, 32);
            const mat  = new THREE.MeshPhongMaterial({ color: colorHex, emissive: emissiveHex, emissiveIntensity: 0.5, shininess: 80 });
            const mesh    = new THREE.Mesh(geo, mat);
            const haloGeo = new THREE.SphereGeometry(size * (isMoon ? 2.5 : 1.6), 24, 24);
            const haloMat = new THREE.MeshBasicMaterial({ color: emissiveHex, transparent: true, opacity: isMoon ? 0.25 : 0.15 });
            mesh.add(new THREE.Mesh(haloGeo, haloMat));
            return mesh;
        }

        function createSprite(text, fontSize, bgColor) {
            const canvas = document.createElement('canvas');
            const ctx    = canvas.getContext('2d');
            ctx.font     = `bold ${fontSize}px Segoe UI, sans-serif`;
            const w      = ctx.measureText(text).width;
            canvas.width  = w + 30;
            canvas.height = fontSize + 20;
            ctx.font      = `bold ${fontSize}px Segoe UI, sans-serif`;
            ctx.fillStyle = bgColor;
            ctx.beginPath();
            ctx.roundRect(0, 0, canvas.width, canvas.height, 12);
            ctx.fill();
            ctx.fillStyle    = '#ffffff';
            ctx.textBaseline = 'middle';
            ctx.fillText(text, 15, canvas.height / 2);
            const tex    = new THREE.CanvasTexture(canvas);
            const sprite = new THREE.Sprite(new THREE.SpriteMaterial({ map: tex, transparent: true }));
            sprite.scale.set(canvas.width / 8, canvas.height / 8, 1);
            return sprite;
        }

        const domainsMap  = {};
        parsedCurriculumData.forEach(topicObj => {
            if (!domainsMap[topicObj.domain]) domainsMap[topicObj.domain] = [];
            domainsMap[topicObj.domain].push(topicObj);
        });

        const domainNames  = Object.keys(domainsMap);
        const domainRadius = 140;
        const orbitPivots  = [];
        const nodeMeshes   = [];
        let   hoveredMesh  = null;

        domainNames.forEach((domainName, i) => {
            const clusterColor = palette[i % palette.length];
            const clusterGroup = new THREE.Group();

            const phi   = Math.acos(1 - 2 * (i + 0.5) / domainNames.length);
            const theta = Math.PI * (1 + Math.sqrt(5)) * i;

            const bx0 = domainRadius * Math.cos(theta) * Math.sin(phi);
            const by0 = domainRadius * Math.cos(phi);
            const bz0 = domainRadius * Math.sin(theta) * Math.sin(phi);

            clusterGroup.userData = {
                bx0, by0, bz0,
                ampX: 18 + Math.random() * 22,
                ampY: 14 + Math.random() * 18,
                ampZ: 18 + Math.random() * 22,
                spX:  (0.0008 + Math.random() * 0.0012) * (Math.random() < 0.5 ? 1 : -1),
                spY:  (0.0006 + Math.random() * 0.0010) * (Math.random() < 0.5 ? 1 : -1),
                spZ:  (0.0008 + Math.random() * 0.0012) * (Math.random() < 0.5 ? 1 : -1),
                phX:  Math.random() * Math.PI * 2,
                phY:  Math.random() * Math.PI * 2,
                phZ:  Math.random() * Math.PI * 2,
                rotSpeedY: (0.0005 + Math.random() * 0.0008) * (Math.random() < 0.5 ? 1 : -1),
                type: 'cluster'
            };

            const domainMesh = createNodeMesh(9, clusterColor.hex, clusterColor.em);
            domainMesh.userData = { label: domainName, type: 'domain' };
            clusterGroup.add(domainMesh);
            nodeMeshes.push(domainMesh);

            const dSprite = createSprite(domainName, 44, clusterColor.rgba);
            dSprite.position.set(0, 20, 0);     
            domainMesh.add(dSprite);

            const topics = domainsMap[domainName];

            topics.forEach((topicObj, j) => {
                const planetGroup = new THREE.Group();
                const baseSpeed   = 0.35 - j * 0.04;               
                const pSpeed      = Math.max(0.08, baseSpeed) * (j % 2 === 0 ? 1 : -1) / 300;  
                const pAngle0     = (j / topics.length) * Math.PI * 2;
                const pOrbit      = 35 + j * 8 + Math.random() * 8; 

                const pTiltX = (Math.random() - 0.5) * 0.6;         
                const pTiltZ = (Math.random() - 0.5) * 0.3;         

                planetGroup.userData = {
                    orbitR: pOrbit, angle: pAngle0, speed: pSpeed,
                    tiltX: pTiltX, tiltZ: pTiltZ,
                    type: 'planet'
                };

                const cleanTopicName = topicObj.topic.replace(/^\d+\.\s*/, '');
                const topicMesh      = createNodeMesh(5, clusterColor.hex, clusterColor.em);
                topicMesh.userData   = { label: cleanTopicName, type: 'topic' };
                planetGroup.add(topicMesh);
                nodeMeshes.push(topicMesh);

                const tSprite = createSprite(cleanTopicName, 32, clusterColor.rgba);
                tSprite.position.set(0, 11, 0);
                tSprite.visible = false;
                topicMesh.add(tSprite);

                const orbitPts  = new THREE.Path().absarc(0, 0, pOrbit, 0, Math.PI * 2).getSpacedPoints(64);
                const orbitGeo  = new THREE.BufferGeometry().setFromPoints(orbitPts);
                const orbitLine = new THREE.Line(orbitGeo, new THREE.LineBasicMaterial({ color: clusterColor.hex, transparent: true, opacity: 0.15 }));
                orbitLine.rotation.x = Math.PI / 2 + pTiltX;
                orbitLine.rotation.z = pTiltZ;
                clusterGroup.add(orbitLine);

                topicObj.subjects.forEach((sub, k) => {
                    const moonGroup = new THREE.Group();
                    const mSpeed  = (0.55 + k * 0.08) * (k % 2 === 0 ? 1 : -1) / 360;
                    const mAngle0 = (k / topicObj.subjects.length) * Math.PI * 2;
                    const mOrbit  = 14 + k * 3 + Math.random() * 4;  

                    const mTiltX  = (Math.random() - 0.5) * 0.45;    
                    const mTiltZ  = (Math.random() - 0.5) * 0.2;

                    moonGroup.userData = {
                        orbitR: mOrbit, angle: mAngle0, speed: mSpeed,
                        tiltX: mTiltX, tiltZ: mTiltZ,
                        type: 'moon'
                    };

                    const subMesh    = createNodeMesh(2.5, clusterColor.hex, clusterColor.em, true);
                    subMesh.userData = { label: sub.name, type: 'subject', fullObj: sub, color: clusterColor.base };
                    moonGroup.add(subMesh);
                    nodeMeshes.push(subMesh);

                    const sSprite = createSprite(sub.name, 24, 'rgba(0,0,0,0.6)');
                    sSprite.position.set(0, 6, 0);
                    sSprite.visible = false;
                    subMesh.add(sSprite);

                    const mOrbitPts  = new THREE.Path().absarc(0, 0, mOrbit, 0, Math.PI * 2).getSpacedPoints(32);
                    const mOrbitGeo  = new THREE.BufferGeometry().setFromPoints(mOrbitPts);
                    const mOrbitLine = new THREE.Line(mOrbitGeo, new THREE.LineBasicMaterial({ color: 0x334155, transparent: true, opacity: 0.2 }));
                    mOrbitLine.rotation.x = Math.PI / 2 + mTiltX;
                    mOrbitLine.rotation.z = mTiltZ;
                    planetGroup.add(mOrbitLine);
                    planetGroup.add(moonGroup);
                });

                clusterGroup.add(planetGroup);
            });

            orbitPivots.push(clusterGroup);
            scene.add(clusterGroup);
        });

        const raycaster = new THREE.Raycaster();
        const mouse     = new THREE.Vector2();
        const tooltip   = document.getElementById('tooltip');

        container.addEventListener('mousemove', e => {
            mouse.x = (e.clientX / window.innerWidth) * 2 - 1;
            mouse.y = -(e.clientY / (window.innerHeight - 60)) * 2 + 1;
            raycaster.setFromCamera(mouse, camera);
            const hits = raycaster.intersectObjects(nodeMeshes);

            if (hoveredMesh) {
                if (hoveredMesh.userData.type !== 'domain') {
                    const sprite = hoveredMesh.children.find(c => c.type === 'Sprite');
                    if (sprite) sprite.visible = false;
                }
                hoveredMesh = null;
            }

            if (hits.length) {
                hoveredMesh = hits[0].object;
                const ud    = hoveredMesh.userData;
                const prefix = ud.type === 'domain'  ? '🌟 ' : ud.type === 'topic'   ? '🪐 ' : '🌙 ';
                tooltip.innerHTML      = prefix + ud.label;
                tooltip.style.left     = (e.clientX + 14) + 'px';
                tooltip.style.top      = (e.clientY + 14) + 'px';
                tooltip.style.display  = 'block';

                container.style.cursor = ud.type === 'subject' ? 'pointer' : 'default';

                if (ud.type !== 'domain') {
                    const sprite = hoveredMesh.children.find(c => c.type === 'Sprite');
                    if (sprite) sprite.visible = true;
                }
            } else {
                tooltip.style.display  = 'none';
                container.style.cursor = 'default';
            }
        });

        container.addEventListener('click', e => {
            raycaster.setFromCamera(mouse, camera);
            const hits = raycaster.intersectObjects(nodeMeshes);
            if (hits.length && hits[0].object.userData.type === 'subject') {
                openModal(hits[0].object.userData.fullObj, hits[0].object.userData.color);
            }
        });

        container.addEventListener('dblclick', e => {
            raycaster.setFromCamera(mouse, camera);
            const hits = raycaster.intersectObjects(nodeMeshes);
            if (!hits.length) {
                const startPos = camera.position.clone();
                const target   = new THREE.Vector3(homePosition.x, homePosition.y, homePosition.z);
                let   t        = 0;
                function resetAnim() {
                    t += 0.04;
                    camera.position.lerpVectors(startPos, target, Math.min(t, 1));
                    controls.update();
                    if (t < 1) requestAnimationFrame(resetAnim);
                }
                resetAnim();
            }
        });

        container.addEventListener('touchstart', e => {
            if (e.touches.length === 1) {
                mouse.x = (e.touches[0].clientX / window.innerWidth) * 2 - 1;
                mouse.y = -(e.touches[0].clientY / (window.innerHeight - 60)) * 2 + 1;
                raycaster.setFromCamera(mouse, camera);
                const hits = raycaster.intersectObjects(nodeMeshes);
                if (hits.length && hits[0].object.userData.type === 'subject') {
                    openModal(hits[0].object.userData.fullObj, hits[0].object.userData.color);
                }
            }
        }, { passive: true });

        const hint = document.getElementById('hint');
        if ('ontouchstart' in window) {
            hint.textContent = '👆 Tap moon to explore • Pinch to zoom • Drag to rotate';
        }
        setTimeout(() => {
            hint.style.transition = 'opacity 1.5s ease';
            hint.style.opacity    = '0';
            setTimeout(() => { hint.style.pointerEvents = 'none'; }, 1500);
        }, 4000);

        let time = 0;
        function animate() {
            requestAnimationFrame(animate);
            time++;

            starField.rotation.y = time * 0.00008;

            orbitPivots.forEach(cl => {
                const ud = cl.userData;
                cl.position.x = ud.bx0 + Math.sin(time * ud.spX + ud.phX) * ud.ampX;
                cl.position.y = ud.by0 + Math.cos(time * ud.spY + ud.phY) * ud.ampY; 
                cl.position.z = ud.bz0 + Math.sin(time * ud.spZ + ud.phZ) * ud.ampZ;
                cl.rotation.y += ud.rotSpeedY;

                cl.children.forEach(child => {
                    if (!child.userData) return;
                    if (child.userData.type === 'planet') {
                        child.userData.angle += child.userData.speed;
                        const a    = child.userData.angle;
                        const r    = child.userData.orbitR;
                        const tx   = child.userData.tiltX;

                        child.position.x = Math.cos(a) * r;
                        child.position.y = Math.sin(a) * r * Math.sin(tx); 
                        child.position.z = Math.sin(a) * r * Math.cos(tx);

                        child.children.forEach(grandchild => {
                            if (!grandchild.userData) return;
                            if (grandchild.userData.type === 'moon') {
                                grandchild.userData.angle += grandchild.userData.speed;
                                const ma   = grandchild.userData.angle;
                                const mr   = grandchild.userData.orbitR;
                                const mtx  = grandchild.userData.tiltX;

                                grandchild.position.x = Math.cos(ma) * mr;
                                grandchild.position.y = Math.sin(ma) * mr * Math.sin(mtx);
                                grandchild.position.z = Math.sin(ma) * mr * Math.cos(mtx);
                            }
                        });
                    }
                });
            });

            controls.update();
            renderer.render(scene, camera);
        }
        animate();

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / (window.innerHeight - 60);
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight - 60);
        });
    }

    // ─────────────────────────────────────────────────────────────────
    // VIEW 2: MINDMAP
    // ─────────────────────────────────────────────────────────────────
    function initMindmap() {
        const container = document.getElementById('mindmap-container');
        const courseTitle = extractCourseTitle(curriculumMarkdown);

        const nodesDataSet = new vis.DataSet([
            { id: 0, label: courseTitle, shape: "star", size: 30, color: { background: "#38bdf8", border: "#0284c7" }, font: { color: "white", size: 16, bold: true }, _type: "center" }
        ]);

        const edges = [];
        let nodeId = 1;

        let currentDomainId    = null;  
        let currentTopicId     = null;  
        let currentDomainColor = null;  
        let domainColorIndex   = 0;     

        const lines = curriculumMarkdown.split("\n");

        for (const line of lines) {
            const trimmed = line.trim();
            if (!trimmed) continue;
            if (trimmed.startsWith("#### ")) continue;

            else if (trimmed.startsWith("### ")) {
                const label = trimmed.slice(4).trim();
                const topicColor = currentDomainColor || "#22c55e";

                nodesDataSet.add({
                    id: nodeId,
                    label: label,
                    shape: "ellipse",
                    size: 14,                           
                    color: { background: topicColor + "bb", border: topicColor },
                    font: { color: "white", size: 11 },
                    _type: "topic"
                });

                if (currentDomainId !== null) {
                    edges.push({ from: currentDomainId, to: nodeId, length: 140, width: 1.5 });
                }
                currentTopicId = nodeId;
                nodeId++;
            }
            else if (trimmed.startsWith("## ")) {
                const label = trimmed.slice(3).trim();
                currentDomainColor = palette[domainColorIndex % palette.length].base;
                domainColorIndex++;

                nodesDataSet.add({
                    id: nodeId,
                    label: label,
                    shape: "dot",
                    size: 20,                           
                    color: { background: currentDomainColor, border: currentDomainColor },
                    font: { color: "white", size: 14, bold: true },
                    _type: "domain"
                });

                edges.push({ from: 0, to: nodeId, length: 200, width: 2 });
                currentDomainId    = nodeId;
                currentTopicId     = null;              
                nodeId++;
            }
            else if (trimmed.startsWith("# ")) {
                continue;                               
            }
            else if (trimmed.startsWith("- ")) {
                const fullText  = trimmed.slice(2).trim();  
                const colonIdx  = fullText.indexOf(":");
                const moonLabel = colonIdx > -1 ? fullText.slice(0, colonIdx).trim() : fullText;                             
                const moonBg    = (currentDomainColor || "#94a3b8") + "66"; 
                const moonBd    = (currentDomainColor || "#94a3b8") + "99"; 

                nodesDataSet.add({
                    id: nodeId,
                    label: moonLabel,
                    shape: "dot",
                    size: 8,                                
                    color: { background: moonBg, border: moonBd },
                    font: { color: "#cbd5e1", size: 9 },
                    title: fullText,                         
                    _type: "moon"
                });

                if (currentTopicId !== null) {
                    edges.push({ from: currentTopicId, to: nodeId, length: 80, width: 1, dashes: true });
                }
                nodeId++;
            }
        } 

        const edgesDataSet = new vis.DataSet(edges);

        networkInstance = new vis.Network(
            container,
            { nodes: nodesDataSet, edges: edgesDataSet },
            {
                layout: { randomSeed: 42 },             
                physics: {
                    solver: "forceAtlas2Based",
                    forceAtlas2Based: { gravitationalConstant: -80, centralGravity: 0.005, springLength: 120, springConstant: 0.04, damping: 0.4 },
                    stabilization: { iterations: 300, fit: true }
                },
                edges: {
                    smooth: { type: "cubicBezier", roundness: 0.5 },
                    color: { color: "#475569", highlight: "#38bdf8", opacity: 0.6 }
                },
                nodes: {
                    borderWidth: 1.5,
                    shadow: { enabled: true, size: 8, x: 2, y: 2, color: "rgba(0,0,0,0.4)" }
                },
                interaction: { hover: true, tooltipDelay: 80, zoomView: true }
            }
        );

        networkInstance.on("click", function(params) {
            if (params.nodes.length > 0) {
                selectedMindmapNodeId = params.nodes[0];
                
                const connectedEdges = networkInstance.getConnectedEdges(selectedMindmapNodeId);
                edgesDataSet.forEach(edge => {
                    if (connectedEdges.includes(edge.id)) {
                        edgesDataSet.update({ id: edge.id, color: { color: "#38bdf8", opacity: 1.0 }, width: 2.5 });
                    } else {
                        edgesDataSet.update({ id: edge.id, color: { color: "#334155", opacity: 0.25 }, width: 1 });
                    }
                });
            } else {
                selectedMindmapNodeId = null;
                edgesDataSet.forEach(edge => {
                    edgesDataSet.update({ id: edge.id, color: { color: "#475569", opacity: 0.6 }, width: 1.5 });
                });
            }
        });

        networkInstance.on("afterDrawing", (ctx) => {
            if (mapAnimationProfile !== 'pulse') return;

            const edgeData = networkInstance.body.edges;

            ctx.save();
            for (const id in edgeData) {
                const edge = edgeData[id];

                // Direct click traversal restriction matching focus targets
                if (selectedMindmapNodeId !== null && edge.fromId !== selectedMindmapNodeId && edge.toId !== selectedMindmapNodeId) {
                    continue; 
                }

                // SECURE FIX: Grab coordinate endpoints directly out of vis rendering engine cache!
                if (!edge.edgeType || !edge.edgeType.from || !edge.edgeType.to) continue;

                const startX = edge.edgeType.from.x;
                const startY = edge.edgeType.from.y;
                const endX = edge.edgeType.to.x;
                const endY = edge.edgeType.to.y;

                let currentPulseX, currentPulseY;
                const t = pulseProgress;

                // Accurately extract control anchors used by the cubicBezier module
                if (typeof edge.edgeType.getBezierPoints === 'function') {
                    const points = edge.edgeType.getBezierPoints();
                    if (points && points.length === 4) {
                        // Standard parametric formula computing true cubic bends
                        currentPulseX = Math.pow(1 - t, 3) * points[0].x + 3 * Math.pow(1 - t, 2) * t * points[1].x + 3 * (1 - t) * Math.pow(t, 2) * points[2].x + Math.pow(t, 3) * points[3].x;
                        currentPulseY = Math.pow(1 - t, 3) * points[0].y + 3 * Math.pow(1 - t, 2) * t * points[1].y + 3 * (1 - t) * Math.pow(t, 2) * points[2].y + Math.pow(t, 3) * points[3].y;
                    }
                }

                // Seamless arithmetic linear calculation fallback 
                if (currentPulseX === undefined) {
                    currentPulseX = startX + (endX - startX) * t;
                    currentPulseY = startY + (endY - startY) * t;
                }

                ctx.beginPath();
                const gradient = ctx.createRadialGradient(currentPulseX, currentPulseY, 0.5, currentPulseX, currentPulseY, 4);
                gradient.addColorStop(0, '#ffffff');
                gradient.addColorStop(0.3, '#38bdf8');
                gradient.addColorStop(1, 'rgba(56, 189, 248, 0)');
                
                ctx.fillStyle = gradient;
                ctx.arc(currentPulseX, currentPulseY, 4, 0, Math.PI * 2);
                ctx.fill();
            }
            ctx.restore();
        });

        setMapAnimation('pulse');
    }

    function setMapAnimation(profileMode) {
        mapAnimationProfile = profileMode;
        
        document.querySelectorAll('.anim-btn').forEach(b => b.classList.remove('active-mode'));
        const activeBtn = document.getElementById(`btn-anim-${profileMode}`);
        if (activeBtn) activeBtn.classList.add('active-mode');
        
        if (!networkInstance) return;

        if (mindmapLoopId) { cancelAnimationFrame(mindmapLoopId); mindmapLoopId = null; }

        if (profileMode === 'static') {
            networkInstance.setOptions({ physics: { enabled: false } });
        } else if (profileMode === 'drift') {
            networkInstance.setOptions({
                physics: {
                    enabled: true,
                    stabilization: false, 
                    solver: "forceAtlas2Based",
                    forceAtlas2Based: { gravitationalConstant: -55, centralGravity: 0.002, springLength: 135, springConstant: 0.01, damping: 0.04 }
                }
            });
            runMindmapDriftAwakeLoop();
        } else if (profileMode === 'pulse') {
            networkInstance.setOptions({ 
                physics: { 
                    enabled: true, 
                    stabilization: false,
                    solver: "forceAtlas2Based",
                    forceAtlas2Based: { gravitationalConstant: -70, centralGravity: 0.005, springLength: 125, springConstant: 0.03, damping: 0.35 }
                } 
            });
            runMindmapPulseLoop();
        }
    }

    function runMindmapPulseLoop() {
        if (mapAnimationProfile !== 'pulse' || !networkInstance) {
            mindmapLoopId = null;
            return;
        }
        
        pulseProgress += 0.004; 
        if (pulseProgress > 1) {
            pulseProgress = 0; 
        }
        
        networkInstance.startSimulation();
        networkInstance.redraw();
        mindmapLoopId = requestAnimationFrame(runMindmapPulseLoop);
    }

    function runMindmapDriftAwakeLoop() {
        if (mapAnimationProfile !== 'drift' || !networkInstance) {
            mindmapLoopId = null;
            return;
        }
        networkInstance.startSimulation();
        mindmapLoopId = requestAnimationFrame(runMindmapDriftAwakeLoop);
    }

    function initCards() {
        const grid = document.getElementById("grid");
        const search = document.getElementById("search");
        const domainFilter = document.getElementById("domainFilter");

        const domains = [...new Set(parsedCurriculumData.map(d => d.domain))];
        domains.forEach(d => {
            const opt = document.createElement("option");
            opt.value = d;
            opt.textContent = d;
            domainFilter.appendChild(opt);
        });

        window.filterCards = function () {
            const q = search.value.toLowerCase();
            const filterDomain = domainFilter.value;
            grid.innerHTML = "";
            let matchCount = 0;

            parsedCurriculumData.forEach(topicObj => {
                if (filterDomain !== "all" && topicObj.domain !== filterDomain) return;

                const domainIdx = domains.indexOf(topicObj.domain);
                const clusterColor = palette[domainIdx % palette.length].base;
                const cleanTopicName = topicObj.topic.replace(/^\d+\.\s*/, '');

                topicObj.subjects.forEach(sub => {
                    const searchString = `${sub.name} ${cleanTopicName} ${sub.desc}`.toLowerCase();
                    if (searchString.includes(q)) {
                        matchCount++;
                        const card = document.createElement("div");
                        card.className = "card";
                        card.style.borderLeft = `4px solid ${clusterColor}`;
                        card.innerHTML = `
                            <div class="id" style="color:${clusterColor}">${sub.id}</div>
                            <h3 style="color:var(--text)">${sub.name}</h3>
                            <p>${sub.desc}</p>
                            <span class="phase-tag" style="color:${clusterColor};background:${clusterColor}20">
                                ${topicObj.domain} • ${cleanTopicName}
                            </span>`;
                        card.onclick = () => openModal(sub, clusterColor);
                        grid.appendChild(card);
                    }
                });
            });

            if (matchCount === 0) {
                grid.innerHTML = '<div class="empty">No matching subjects found. Try another search.</div>';
            }
        };

        search.addEventListener("input", window.filterCards);
        domainFilter.addEventListener("change", window.filterCards);
        window.filterCards();
    }

    // ─────────────────────────────────────────────────────────────────
    // VIEW 4: UNIVERSE AT A GLANCE
    // ─────────────────────────────────────────────────────────────────
    function initVisual() {
        const grid = document.getElementById("visual-grid");
        let numTopics = 0;
        let numSubs = 0;

        const domains = [...new Set(parsedCurriculumData.map(d => d.domain))];

        domains.forEach((domainName, idx) => {
            const clusterColor = palette[idx % palette.length].base;
            const clusterBg = palette[idx % palette.length].rgba.replace('0.75', '0.15');

            const topics = parsedCurriculumData.filter(d => d.domain === domainName);
            numTopics += topics.length;

            const pills = topics.map(t => {
                const cleanTopicName = t.topic.replace(/^\d+\.\s*/, '');
                return `<span class="topic-pill" style="color:${clusterColor};background:${clusterBg};border:1px solid ${clusterColor}33">${cleanTopicName}</span>`;
            }).join('');

            const allSubs = topics.flatMap(t => t.subjects);
            numSubs += allSubs.length;

            const tags = allSubs.slice(0, 10).map(s =>
                `<span class="moon-tag" style="background:${clusterColor}15;border:1px solid #334155;color:${clusterColor}">${s.name}</span>`
            ).join('') + (allSubs.length > 10
                ? `<span class="moon-tag" style="background:${clusterColor}15;border:1px solid #334155;color:${clusterColor}">+${allSubs.length - 10} more</span>`
                : '');

            const card = document.createElement('div');
            card.className = 'visual-card';
            card.onclick = () => openDomainDetailModal(domainName);
            card.innerHTML = `
              <div class="visual-card-header">
                <div>
                  <h3 style="color:${clusterColor}">${domainName}</h3>
                  <div style="font-size:0.7rem;color:var(--muted);margin-top:2px">
                    ${topics.length} topics · ${allSubs.length} subjects
                  </div>
                </div>
              </div>
              <div class="topic-pill-list">${pills}</div>
              <div class="moon-tag-list">${tags}</div>`;
            grid.appendChild(card);
        });

        document.getElementById('stat-domains').textContent = domains.length;
        document.getElementById('stat-topics').textContent = numTopics;
        document.getElementById('stat-subs').textContent = numSubs;
    }

    function toggleClearButton(value) {
        document.getElementById('custom-clear-btn').style.display = value.length > 0 ? 'block' : 'none';
    }

    function clearSearch() {
        const input = document.getElementById('search');
        input.value = '';
        toggleClearButton('');
        if (typeof filterCards === 'function') {
            filterCards();
        } else {
            input.dispatchEvent(new Event('input'));
        }
        input.focus();
    }

    document.addEventListener('click', e => {
        const sidePanel = document.getElementById('side-panel');
        
        if (document.body.classList.contains('panel-open')) {
            if (!sidePanel.contains(e.target) && 
                (e.target.closest('.view') || e.target.tagName === 'BODY')) {
                
                if (!e.target.closest('.card') && !e.target.closest('.visual-card') && !e.target.closest('.anim-btn')) {
                    closeSidePanel();
                }
            }
        }
    });
